
```python
rfc = ml_model.tuning(
    model_name = 'random_forest_classifier',
    model_algorithm = RandomForestClassifier(random_state=42), 
    param_grid = { 
        'n_estimators': [200, 500],
        'max_features': ['auto', 'sqrt'],
        'max_depth' : [4,5,100],
        'criterion' :['gini', 'entropy']
    }, 
    folds = 5, 
    grid_search = True,
    best_estimator = True
)
```

```python
ml_model.tp_rate_analysis(models=(lrc,rfc))
```

